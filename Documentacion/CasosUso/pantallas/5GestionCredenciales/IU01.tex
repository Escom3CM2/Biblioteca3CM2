\subsection{IU01 Pantalla de Renovacion}

\subsubsection{Objetivo}
	Pantalla donde el bibliotecario ingresa el id del lector.

\subsubsection{Diseño}
	Esta pantalla aparece al dar click en el apartado Renovación de Credencial. 

\IUfig[.7]{5GestionCredenciales/IU01}{IU01}{Renovar Credencial.}

\subsubsection{Salidas}
ninguna	
	
\subsubsection{Entradas}
	\begin{Citemize}
		\item id de la credencial		
	\end{Citemize}

\subsubsection{Comandos}
\begin{itemize}
	\item \IUbutton{Renovar}: Envía el id del lector para generar el PDF con la credencial renovada.
\end{itemize}
